# GPU-only extras for the editor_gpu render pipeline.
#
# Pulled by Colab + Kaggle workers on top of the base requirements.txt.
# Workers without these (e.g. HF Space CPU container) keep using the
# existing ffmpeg-only path; modules/editor.py's import of editor_gpu
# is guarded by try/except ImportError.
#
# Pin choices:
#   torch 2.4.0 — Colab T4 has CUDA 12.x; the cu121 wheel matches.
#   decord 0.6.0 — only release with stable NVDEC GPU decode support.
#   av 12.3.0 — PyAV with h264_nvenc encoder bindings; older wheels
#               (<12) are missing the CUDA HW context plumbing.

--extra-index-url https://download.pytorch.org/whl/cu121
torch==2.4.0+cu121
decord==0.6.0
av==12.3.0

# Kokoro TTS — high-quality neural voice model that runs on the GPU
# workers. modules/voiceover.py tries Kokoro first for English and
# falls back to edge-tts if the import fails. Skipping this line
# means every render logs a 'Kokoro not available' warning and uses
# the lower-quality edge-tts voice.
kokoro>=0.9.4
misaki>=0.9.4
soundfile>=0.12
# Local SDXL + Flux 2 klein-4B image generation via diffusers. Providers
# are opt-in via the /settings "AI image generation" card and no-op on
# CPU-only workers (checks torch.cuda.is_available() at load time).
#
# Version pins:
#   - diffusers >= 0.36 ships the Flux2Pipeline class needed for klein-4B.
#     0.30-0.35 don't have it and would fail at
#     `from diffusers import Flux2KleinPipeline`.
#   - transformers >= 4.51 is required by the Qwen3 text encoder klein-4B
#     bundles. Older transformers throws
#     `Expected Mistral3ForConditionalGeneration, got Qwen3ForCausalLM`.
#   - Kokoro pins transformers < 4.50 in older releases. Kaggle notebook
#     cell 2 tries `kokoro >= 1.0` first; if that resolver-conflicts,
#     it installs Kokoro with --no-deps and voiceover.py falls back to
#     edge-tts at runtime (already implemented — Oracle uses this path
#     today).
#   - sentencepiece is an indirect dep of some diffusers builds when
#     Qwen3's tokenizer needs SP fallback.
diffusers>=0.36
transformers>=4.51
accelerate>=0.30
sentencepiece>=0.2
# Kokoro 0.9+ pulls `misaki` for phoneme handling. misaki calls
# EspeakWrapper.set_data_path(...) which only exists in phonemizer-fork
# (vanilla `phonemizer` on PyPI stopped receiving updates in 2023).
# espeakng-loader ships the data path misaki asks for. All three are
# required — omitting any one raises `AttributeError: type object
# 'EspeakWrapper' has no attribute 'set_data_path'` at Kokoro import.
phonemizer-fork>=3.3.2
espeakng-loader>=0.2
