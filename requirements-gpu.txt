# GPU-only extras for the editor_gpu render pipeline.
#
# Pulled by Colab + Kaggle workers on top of the base requirements.txt.
# Workers without these (e.g. HF Space CPU container) keep using the
# existing ffmpeg-only path; modules/editor.py's import of editor_gpu
# is guarded by try/except ImportError.
#
# Pin choices:
#   torch 2.4.0 — Colab T4 has CUDA 12.x; the cu121 wheel matches.
#   decord 0.6.0 — only release with stable NVDEC GPU decode support.
#   av 12.3.0 — PyAV with h264_nvenc encoder bindings; older wheels
#               (<12) are missing the CUDA HW context plumbing.

--extra-index-url https://download.pytorch.org/whl/cu121
torch==2.4.0+cu121
decord==0.6.0
av==12.3.0

# Kokoro TTS — high-quality neural voice model that runs on the GPU
# workers. modules/voiceover.py tries Kokoro first for English and
# falls back to edge-tts if the import fails. Skipping this line
# means every render logs a 'Kokoro not available' warning and uses
# the lower-quality edge-tts voice.
kokoro>=0.9.4
misaki>=0.9.4
soundfile>=0.12
# Local SDXL + klein-4B image generation via diffusers.
#
# CAPS matter: 2026-07-13 boot log showed pip resolving
# `transformers>=4.55` to transformers 5.13.1 (major rewrite that
# dropped AlbertModel + BloomPreTrainedModel and moved Qwen3ForCausalLM
# — broke Kokoro + diffusers AutoPipeline + klein-4B all at once).
# Same class of runaway happened with diffusers landing on 0.39.0.
# Pin both to their 4.x / 0.3x lines.
#
#   diffusers 0.36..<0.40 ships Flux2KleinPipeline; 0.39 works.
#   transformers 4.55..<5.0 has Qwen3ForCausalLM; 5.x removed it.
#   sentencepiece is required by Qwen3's tokenizer fallback.
#   Kokoro will still fall back to edge-tts on this transformers
#     version (its AlbertModel path breaks at 4.50+). voiceover.py
#     handles that gracefully at runtime.
# 2026-07-13 attempt 4 pin narrowing.
# Attempt 3 landed transformers 4.57.6 + diffusers 0.39.0. Both had
# broken top-level exports:
#   cannot import name 'AlbertModel'         from 'transformers'
#   cannot import name 'BloomPreTrainedModel' from 'transformers'
#   cannot import name 'Qwen3ForCausalLM'    from 'transformers'
# HuggingFace deprecated top-level model exports around transformers
# 4.56 (submodule-only from that release onward). Kokoro AND
# diffusers 0.39 both do `from transformers import <X>` at import
# time and blow up.
# Fix window:
#   transformers 4.51..<4.56 — Qwen3ForCausalLM added, top-level
#     exports still intact.
#   diffusers 0.36..<0.37    — first Flux2KleinPipeline release; later
#     0.37/0.38/0.39 track newer transformers layouts.
# If 4.55.x still exports Qwen3ForCausalLM, klein-4B works AND Kokoro
# might survive too.
diffusers>=0.36,<0.37
transformers>=4.51,<4.56
accelerate>=0.30,<2.0
sentencepiece>=0.2
# Kokoro 0.9+ pulls `misaki` for phoneme handling. misaki calls
# EspeakWrapper.set_data_path(...) which only exists in phonemizer-fork
# (vanilla `phonemizer` on PyPI stopped receiving updates in 2023).
# espeakng-loader ships the data path misaki asks for. All three are
# required — omitting any one raises `AttributeError: type object
# 'EspeakWrapper' has no attribute 'set_data_path'` at Kokoro import.
phonemizer-fork>=3.3.2
espeakng-loader>=0.2
